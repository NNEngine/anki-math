\section*{Jacobian Matrix and Determinant}

\subsection*{Basic Definitions}

\begin{enumerate}

\item Let $F : \mathbb{R}^n \to \mathbb{R}^m$ be given by
  \[
    F(x_1,\dots,x_n) = (f_1(x),\dots,f_m(x)).
  \]
  What is the definition of the \emph{Jacobian matrix} $J_F(x)$?

\item Write explicitly the Jacobian matrix:
  \[
    J_F(x) = \left( \frac{\partial f_i}{\partial x_j} \right).
  \]
  What are its dimensions?

\item In which case is the Jacobian matrix square?

\item When $m = n$, how is the \emph{Jacobian determinant} defined?

\item What does the notation
  \[
    \det J_F(x)
  \]
  represent?

\end{enumerate}

\subsection*{Change of Variables in Integration}

\begin{enumerate}

\item State the change-of-variables formula in $\mathbb{R}^n$:
  \[
    \int_{F(U)} g(y)\,dy
    =
    \int_U g(F(x))\, |\det J_F(x)| \, dx.
  \]

\item Why is the absolute value necessary?

\item What conditions must $F$ satisfy for the change-of-variables formula to hold?

\item Compute the Jacobian determinant for the transformation from Cartesian to polar coordinates:

  \]

\item Why does the area element become
  \[
    dx\,dy = r\,dr\,d\theta?
  \]

\end{enumerate}
